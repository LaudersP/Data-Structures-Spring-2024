249 void breadthFirst(ssuds::ArrayListV2<std::pair<N, N>>& traversalMap, N startNode) {
250 // Check that startNode is valid
251 if (!contains_node(startNode))
252 throw std::out_of_range("ERROR: Start node does not exist in the graph!\n");
253
254 ssuds::Queue<N> frontier;
255 ssuds::UnorderedMap<N, bool> visited;
256 traversalMap.clear();
257
258 frontier.enqueue(startNode);
259 visited[startNode] = true;
260 traversalMap.append(std::make_pair(startNode, N()));
261
262 // Loop through the queue
263 while (!frontier.empty()) {
264 N current = frontier.dequeue();
265 typename UnorderedMap<N, UnorderedMap<N, E>>::UnorderedMapIterator it = mData.find(current);
266
267 // Check that the iterator is valid
268 if (it != mData.end()) {
269 // Iterate through the current nodes edges
270 for (std::pair<N, E>& edges : (*it).second) {
271 N neighbor = edges.first;
272
273 // Check if that node has been visited or not
274 if (!visited[neighbor]) {
275 visited[neighbor] = true;
276 frontier.enqueue(neighbor);
277 traversalMap.append(std::make_pair(neighbor, current));
278 }
279 }
280 }
281 }
