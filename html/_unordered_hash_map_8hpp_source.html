194 V& operator[](const K& key) {
195 size_t index = find_spot(key);
196 if (mData[index] == NULL) {
197 // That value isn't in here. Create it and return
198 // a reference to the new node's value.

209 // We do have that item. Return it to them -- they

280 V remove(const K& key) {
281 size_t index = find_spot(key);
282 size_t start_index = index;
283 bool found_it = false;
284 V found_val;
285 bool found_null = false;
286
287 // First, collect all the values we find after that index until we
288 // find a NULL.
289 ArrayListV2<K> keys_to_readd;
290 ArrayListV2<V> values_to_readd;
291
292 while (true) {
293 if (mData[index] == NULL) {
294 found_null = true;
295 break;
296 }
297 else {
298 if (mData[index]->first == key) {
299 found_val = mData[index]->second;
300 found_it = true;
301 }
302 else {
303 keys_to_readd.append(mData[index]->first);
304 values_to_readd.append(mData[index]->second);
305 }
306 delete mData[index];
307 mData[index] = NULL;
308 }
309 ++index;
310 if (index == start_index)
311 throw std::out_of_range("invalid heap state");
312 else if (index == mCapacity)
313 index = 0;
314 }
315
316 // Now, re-add any key-value pairs that we removed along the way
317 mSize -= keys_to_readd.size() + 1;
318 for (int i = 0; i < (int)keys_to_readd.size(); i++)
319 (*this)[keys_to_readd[i]] = values_to_readd[i];
320
321 // Return the removed element
322 if (found_it)
323 return found_val;
324 else
325 throw std::out_of_range("The value doesn't exist in the map");
326 }

360 size_t find_spot(const K& key) const {
361 size_t start_index = mHashGenerator(key) % mCapacity;
362 size_t cur_index = start_index;
363 for (; cur_index < (int)mCapacity; cur_index++) {
364 if (mData[cur_index] == NULL || mData[cur_index]->first == key)
365 return cur_index;
366 }
367
368 // If we get here, we didn't find it yet, loop back to the beginning
369 for (cur_index = 0; cur_index < start_index; cur_index++) {
370 if (mData[cur_index] == NULL || mData[cur_index]->first == key)
371 return cur_index;
372 }
373
374 // If we get here, the grow operation is not working properly
375 throw std::out_of_range("Invalid state: empty hash table");
376 }

380 bool check_for_capacity() {
381 float cur_capacity = (float)mSize / mCapacity;
382 if (cur_capacity > MAX_LOAD_FACTOR) {
383 // Collect all the key-value pairs
384 ArrayListV2<std::pair<K, V>*> pairs;
385 for (int i = 0; i < mCapacity; i++) {
386 if (mData[i] != NULL)
387 pairs.append(mData[i]);
388 }
389
390 // Re-allocate the array (doubling)
391 mCapacity <<= 1; // bit-shift one place (double)
392 delete[] mData;
393 mData = new std::pair<K, V>* [mCapacity];
394 memset(mData, 0, sizeof(std::pair<K, V>*) * mCapacity);
395
396 // Re-add all the key-value pairs
397 for (int i = 0; i < (int)pairs.size(); i++)
398 internal_add(pairs[i]);
399
400 return true;
401 }
402 else
403 return false;
404 }
